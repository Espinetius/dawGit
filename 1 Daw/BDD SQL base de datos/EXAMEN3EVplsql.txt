/EJERCICIO 2
contar cuantas F tienen dos palabras
set serveroutput on;
declare 
  palabra varchar2(30);
  palabra2 varchar2(30);
  numero number:=0;
  resultado varchar2(30);
begin
  palabra:=&PALABRA;
  for i in 1..length(palabra)
  loop
    resultado:=substr(palabra,i,1);
    if(resultado = 'f') then
      numero := numero + 1;
    end if;
  end loop;
  dbms_output.put_line(numero);
  palabra2:=&IntroducePalabra;
  numero:=0;
  for i in 1..length(palabra2)
  loop
    resultado:=substr(palabra2,i,1);
    if(resultado = 'f')then
      numero:=numero+1;
    end if;
  end loop;
  dbms_output.put_line(numero);
end;

/EJERCICIO 4
3 ARRAYS UNO PARA EQUIPOS, PUNTOS, CIUDADES
SACAR EL NUMERO DE EQUIPOS
NUMERO DE PUNTOS
NUMERO DE CIUDADES
IMPRIMIR CADA EQUIPO CON SUS PUNTOS Y SUS CIUDADES
CUANTOS EQUIPOS QUE SEAN DE MADRID
CUAL ES EL QUE MENOS PUNTOS TIENE

DECLARE
TYPE EQUIPOS IS VARRAY(6) OF VARCHAR2(10);
TYPE PUNTOS IS VARRAY(6) OF NUMBER(2);
TYPE CIUDADES IS VARRAY(6) OF VARCHAR2(10);
EQUIPO EQUIPOS;
PUNTO PUNTOS;
CIUDAD CIUDADES;
CONTADOR NUMBER;
MENOR NUMBER:=99;
PRINGAO VARCHAR2(10);

BEGIN
	EQUIPO:=EQUIPOS('ATLETI','REALMADRID','BARSA','BETIS','RAYO','SPORTING');
	PUNTO:=PUNTOS(20,1,25,22,10,4);
	CIUDAD:=CIUDADES('MADRID','MADRID','BARCELONA','SEVILLA','MADRID','ASTURIAS');
/*CUANTOS EQUIPOS PUNTOS Y CIUDADES HAY*/
DBMS_OUTPUT.PUT_LINE('HAY '||EQUIPO.COUNT||' EQUIPOS');
DBMS_OUTPUT.PUT_LINE('PUNTOS ' || PUNTO.COUNT);
DBMS_OUTPUT.PUT_LINE('CIUDADES ' || CIUDAD.COUNT);
/*CADA EQUIPO CON SUS COSAS*/
CONTADOR:=EQUIPO.COUNT;
FOR I IN 1..CONTADOR LOOP
	DBMS_OUTPUT.PUT_LINE('EL EQUIPO '||EQUIPO(I)|| ' TIENE '|| PUNTO(I) ||' PUNTOS Y ES DE LA CIUDAD ' || CIUDAD(I));
END LOOP;
/*QUE EQUIPOS SON DE MADRID*/
FOR I IN 1..CONTADOR LOOP
	IF CIUDAD(I)='MADRID' THEN
	DBMS_OUTPUT.PUT_LINE(EQUIPO(I)||' ES DE MADRID');
	END IF;
END LOOP;
/*QUE EQUIPO TIENE MENOS PUNTOS*/
FOR I IN 1..CONTADOR LOOP
	IF PUNTO(I)<MENOR THEN
	MENOR:=PUNTO(I);
	PRINGAO:=EQUIPO(I);
	END IF;
END LOOP;
DBMS_OUTPUT.PUT_LINE('EL EQUIPO CON MENOS PUNTOS ES '|| PRINGAO || ' CON '|| MENOR);
END;
